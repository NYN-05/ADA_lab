#include <stdio.h>
#include <stdlib.h>
#include <time.h>

void swap(long int *a, long int *b) {
	long int tmp = *a;
	*a = *b;
	*b = tmp;
}

void selectionsort(long int arr[], long int n) {
	long int i, j, midx;
	for (i = 0; i < n - 1; i++) {
		midx = i;
		for (j = i + 1; j < n; j++) {
			if (arr[j] < arr[midx]) {
				midx = j;
			}
		}
		swap(&arr[midx], &arr[i]);
	}
}

int main() {
	long int n = 1000;
	double tim1[10];
	printf("Input Size\tSelection Sort Time (ms)\n");

	for (int it = 0; it < 5; it++) {
		long int a[n];
		for (int i = 0; i < n; i++) {
			a[i] = rand() % n + 1;
		}

		clock_t start, end;
		start = clock();
		selectionsort(a, n);
		end = clock();

		tim1[it] = ((double)(end - start)) / CLOCKS_PER_SEC * 1000.0;  // in milliseconds
		printf("%ld\t\t%.3f ms\n", n, tim1[it]);

		n += 1000;
	}

	return 0;
}
