#include <stdio.h>
#define INF 999

int cost[20][20], n;

void read() {
    printf("Enter number of vertices: ");
    scanf("%d", &n);
    printf("Enter weighted matrix (0 for self, 999 for âˆž):\n");
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++) {
            scanf("%d", &cost[i][j]);
            if (i != j && cost[i][j] == 0) cost[i][j] = INF;
        }
}

void floydWarshall() {
    for (int k = 0; k < n; k++)
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (cost[i][k] + cost[k][j] < cost[i][j])
                    cost[i][j] = cost[i][k] + cost[k][j];

    printf("All-pairs shortest path matrix:\n");
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++)
            printf("%d\t", cost[i][j]);
        printf("\n");
    }
}

int main() {
    read();
    floydWarshall();
    return 0;
}
